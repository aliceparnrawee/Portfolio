def count_intervals(counts):
    interval_counts = {}
    for index, value in counts.iteritems():
        hour = index[0]
        minute_range = index[1]
        interval_start = f"{hour:02d}:{minute_range.left:02d}"
        interval_end = f"{hour + minute_range.right // 60:02d}:{minute_range.right % 60:02d}"
        interval = f"{interval_start}-{interval_end}"
        interval_counts[interval] = interval_counts.get(interval, 0) + value
    return interval_counts
